CREATE TABLE `admins` (
  `id` smallint(5) unsigned NOT NULL auto_increment,
  `fecha` int(10) unsigned NOT NULL default '0',
  `nick` varchar(20) NOT NULL default '',
  PRIMARY KEY  (`id`),
  KEY `nick` (`nick`)
)
